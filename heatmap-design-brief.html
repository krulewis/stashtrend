Change 1 — Label utility · budgetUtils.js

Add alongside existing utilities. Import in heatmap component.

export function formatGroupLabel(name, maxLen = 14) {
  const shorts = {
    'Auto & Transportation': 'Auto & Transit',
    'Gifts & Donations': 'Gifts & Don.',
    'Food & Dining': 'Food & Dining',
    'Health & Wellness': 'Health & Wellness',
    'Bills & Utilities': 'Bills & Utils',
  };
  if (shorts[name]) return shorts[name];
  if (name.length <= maxLen) return name;
  const words = name.split(' ');
  let result = '';
  for (const w of words) {
    if ((result + ' ' + w).trim().length > maxLen) break;
    result = (result + ' ' + w).trim();
  }
  return result + '.';
}
