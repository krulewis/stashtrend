// New: ~6-line render function above BudgetChart
+ function MonthTick({ x, y, payload }) {
+ const label = fmtBudgetMonth(payload.value)
+ const now = new Date()
+ const [yr, mo] = payload.value.split('-')
+ const isCurrent = +yr === now.getFullYear() && +mo === now.getMonth() + 1
+ return <text x={x} y={y + 10} textAnchor="middle"
+ fill={isCurrent ? COLOR_ACCENT : AXIS_TICK.fill} fontSize={12}>{label}</text>
+ }

// Change: XAxis tick prop
